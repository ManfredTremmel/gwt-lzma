008    package org.dellroad.lzma.client;
009    
010    import org.dellroad.lzma.client.SevenZip.Compression.LZMA.Encoder;
011    
012    /**
013     * Represents LZMA compression modes, from 1 (fastest compression) to 9 (best compression).
014     */
015    public enum CompressionMode {
016    
017        MODE_1(0, 16, 64,  Encoder.EMatchFinderTypeBT2, 3, 0, 2),
018        MODE_2(0, 20, 64,  Encoder.EMatchFinderTypeBT2, 3, 0, 2),
019        MODE_3(1, 19, 64,  Encoder.EMatchFinderTypeBT4, 3, 0, 2),
020        MODE_4(2, 20, 64,  Encoder.EMatchFinderTypeBT4, 3, 0, 2),
021        MODE_5(2, 21, 128, Encoder.EMatchFinderTypeBT4, 3, 0, 2),
022        MODE_6(2, 22, 128, Encoder.EMatchFinderTypeBT4, 3, 0, 2),
023        MODE_7(2, 23, 128, Encoder.EMatchFinderTypeBT4, 3, 0, 2),
024        MODE_8(2, 24, 255, Encoder.EMatchFinderTypeBT4, 3, 0, 2),
025        MODE_9(2, 25, 255, Encoder.EMatchFinderTypeBT4, 3, 0, 2);
026    
027        private final int algorithm;
028        private final int dictionarySize;
029        private final int fb;
030        private final int matchFinder;
031        private final int lc;
032        private final int lp;
033        private final int pb;
034    
035        private CompressionMode(int algorithm, int dictionarySize, int fb, int matchFinder, int lc, int lp, int pb) {
036            this.algorithm = algorithm;
037            this.dictionarySize = dictionarySize;
038            this.fb = fb;
039            this.matchFinder = matchFinder;
040            this.lc = lc;
041            this.lp = lp;
042            this.pb = pb;
043        }
044    
045        /**
046         * Configure the given {@link Encoder} based on this compression mode.
047         */
048        public void configure(Encoder encoder) {
049            if (!encoder.SetAlgorithm(this.algorithm))
050                throw new RuntimeException("unexpected failure");
051            if (!encoder.SetDictionarySize(1 << this.dictionarySize))
052                throw new RuntimeException("unexpected failure");
053            if (!encoder.SetNumFastBytes(this.fb))
054                throw new RuntimeException("unexpected failure");
055            if (!encoder.SetMatchFinder(this.matchFinder))
056                throw new RuntimeException("unexpected failure");
057            if (!encoder.SetLcLpPb(this.lc, this.lp, this.pb))
058                throw new RuntimeException("unexpected failure");
059        }
060    
061        /**
062         * The the instance corresponding to the integer {@code mode}.
063         *
064         * @param mode value from 1 (fastest) to 9 (best)
065         * @throws IllegalArgumentException if {@code mode} is invalid
066         */
067        public static CompressionMode get(int mode) {
068            switch (mode) {
069            case 1:
070                return MODE_1;
071            case 2:
072                return MODE_2;
073            case 3:
074                return MODE_3;
075            case 4:
076                return MODE_4;
077            case 5:
078                return MODE_5;
079            case 6:
080                return MODE_6;
081            case 7:
082                return MODE_7;
083            case 8:
084                return MODE_8;
085            case 9:
086                return MODE_9;
087            default:
088                throw new IllegalArgumentException("invalid mode " + mode);
089            }
090        }
091    }
