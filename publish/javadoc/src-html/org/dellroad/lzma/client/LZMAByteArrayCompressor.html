008    package org.dellroad.lzma.client;
009    
010    import java.io.ByteArrayInputStream;
011    import java.io.ByteArrayOutputStream;
012    import java.io.IOException;
013    
014    /**
015     * LZMA compressor for {@code byte[]} arrays.
016     */
017    public class LZMAByteArrayCompressor extends LZMACompressor {
018    
019        private final ByteArrayOutputStream output;
020    
021        /**
022         * Construct an encoder using the {@link #DEFAULT_COMPRESSION_MODE default compression mode}.
023         *
024         * <p>
025         * This is a convenience constructor, equivalent to:
026         * <blockquote>
027         * <code>LZMACompressor(data, DEFAULT_COMPRESSION_MODE)</code>
028         * </blockquote>
029         */
030        public LZMAByteArrayCompressor(byte[] data) {
031            this(data, DEFAULT_COMPRESSION_MODE);
032        }
033    
034        /**
035         * Primary constructor.
036         *
037         * @param data uncompressed data
038         * @param mode compression mode
039         * @throws IllegalArgumentException if {@code mode} is null
040         */
041        public LZMAByteArrayCompressor(byte[] data, CompressionMode mode) {
042            this.output = new ByteArrayOutputStream();
043            try {
044                init(new ByteArrayInputStream(data), this.output, data.length, mode);
045            } catch (IOException e) {
046                throw new RuntimeException("impossible exception");
047            }
048        }
049    
050        /**
051         * Get the compressed data.
052         */
053        public byte[] getCompressedData() {
054            return this.output.toByteArray();
055        }
056    }
